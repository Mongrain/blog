写 React / Vue 项目时为什么要在列表组件中写 key，其作用是什么？

2019-08-20

我的回答
因为React和Vue在底层都是用 vNode 来进行diff，这时候如果给每个组件加上一个唯一的 key 。那么下次遇到同样的 key 就可以对组件进行复用。

答案：
vue和react都是采用diff算法来对比新旧虚拟节点，从而进行更新。
在交叉对比中，如果新节点和旧节点的 头尾交叉对比 没有结果时，会根据新节点的 key 去对比旧节点数组中的 key，从而找到相应旧节点（这里对应的是一个key=>index的map映射）。如果没有找到就认为是一个新增节点。而如果没有 key，那么就会采用遍历查找的方式去找到对应的旧节点。一种一个map映射，另一种是遍历查找。相对而言，map 效率更高。

Vue 部分源码

// vue项目  src/core/vdom/patch.js  -488行
// 以下是为了阅读性进行格式化后的代码
// oldCh 是一个旧虚拟节点数组
if (isUndef(oldKeyToIdx)) {
  oldKeyToIdx = createKeyToOldIdx(oldCh, oldStartIdx, oldEndIdx)
}
if(isDef(newStartVnode.key)) {
  // map 方式获取
  idxInOld = oldKeyToIdx[newStartVnode.key]
} else {
  // 遍历方式获取
  idxInOld = findIdxInOld(newStartVnode, oldCh, oldStartIdx, oldEndIdx)
}

创建map函数

function createKeyToOldIdx (children, beginIdx, endIdx) {
  let i, key
  const map = {}
  for (i = beginIdx; i <= endIdx; ++i) {
    key = children[i].key
    if (isDef(key)) map[key] = i
  }
  return map
}

遍历寻找

// sameVnode 是对比新旧节点是否相同的函数
function findIdxInOld (node, oldCh, start, end) {
	for (let i = start; i < end; i++) {
		const c = oldCh[i]

		if (isDef(c) && sameVnode(node, c)) return i
	}
}
